import matplotlib.pyplot as plt 
import seaborn as sns
from pandas import read_csv

df = read_csv('C:\\Users\\immba\\Desktop\\IRABALM GBC\\2ND SEMESTER\\BUS4023\\Week2\\LaptopSalesJanuary2008.csv')
df = df[['Retail Price','Store Postcode']]
df = df[(df['Store Postcode'].isin(['N17 6QA','W4 3PH'])) & (df['Retail Price'])]

x = 'Store Postcode'
y = 'Retail Price'

plt.figure(figsize=(7,7))
ax = sns.boxplot(x=x, y=y, data=df)

plt.suptitle('')
plt.title("Avg. Retail Price per Store")

plt.show()

Based on the boxplot, the distributions are relatively similar. However, average retail prices seem to be slightly higher, on average, in store postcode N17 6QA.
